Genome availability by class

make_shaded_palette() takes the existing per-class colours and generates a lighter/darker shade for each level of a second variable (here: genome available vs. not), so the “available” bars stay visually tied to their class colour.

library(colorspace)


make_shaded_palette <- function(base_cols, shade_labels, amount = 0.3, name_sep=" ", direction=c("light_to_dark", "dark_to_light")) {
  direction <- match.arg(direction)
  n_shades <- length(shade_labels)

  # spread shade amounts evenly from +amount (lightest) to -amount (darkest)
  # e.g. n=2 -> c(0.3, -0.3); n=3 -> c(0.3, 0, -0.3); n=5 -> c(0.3, 0.15, 0, -0.15, -0.3)
  shade_amounts <- seq(amount, -amount, length.out = n_shades)

  if (direction == "dark_to_light") {
    shade_amounts <- rev(shade_amounts)
  }

  shaded <- lapply(names(base_cols), function(nm) {
    sapply(shade_amounts, function(a) {
      if (a > 0) lighten(base_cols[[nm]], amount = a)
      else if (a < 0) darken(base_cols[[nm]], amount = abs(a))
      else base_cols[[nm]]
    })
  })



  shaded_cols <- unlist(shaded)
  names(shaded_cols) <- as.vector(outer(shade_labels, names(base_cols),
