cryozoo_seq_l <- cryozoo_seq %>% 
  mutate(status2=case_when(status%in% "sequenced" & qc_result=="Ok"~ "sequenced",
                           status%in% "sequenced" & qc_result!="Ok"~ "qc_fail",
                           status%in%c("extracted","tech_rep","repeat","ready to process","ready", "pending")~"pending",
                           T~status)) %>% 
  mutate(status2=factor(status2,levels=c("sequenced","submitted","pending","qc_fail"))) %>% 
  select(any_of(c("animal_id", "species", "type","status2","cnag_id"))) %>% 
  group_by(animal_id, species, type) %>% 
  slice_min(status2,with_ties = F) 

cryozoo_seq_w <- cryozoo_seq_l %>% 
  mutate(cnag_id_status=if_else(status2=="qc_fail",paste0(cnag_id," (",status2,")"),cnag_id)) %>%
